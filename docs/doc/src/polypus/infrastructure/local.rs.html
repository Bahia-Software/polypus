1use crate::infrastructure::{QuantumRunner};
2use crate::algorithms::AlgorithmArgs;
3use pyo3::prelude::*;
4use pyo3::types::{PyDict,PyList};
5
6/// LocalRunner struct to run quantum circuits on local infrastructure.
7pub struct LocalRunner;
8
9impl QuantumRunner for LocalRunner {
10
11	/// Run the quantum circuit locally using the provided arguments.
12	fn run(&self, args: &AlgorithmArgs) -> pyo3::PyObject {
13		let id = args.id.clone();
14		let backend = String::from("AerSimulator");
15		let shots = args.shots.clone();
16
17		Python::with_gil(|py| {
18			// build Python list with all qcs
19			let qcs_pylist = PyList::empty(py);
20			for qc in &args.qcs {
21				qcs_pylist.append(qc.clone_ref(py)).unwrap();
22			}
23
24			let module = PyModule::import(py, "polypus_python").unwrap();
25			let connection = module.call_method("connect_to_infrastructure", ("local", ), None);
26			let connection_str = match connection {
27				Ok(conn) => conn.extract::<String>().unwrap(),
28				Err(e) => {
29					panic!("{e}, Error connecting to infrastructure");
30				},
31			};
32			
33			let kwargs = PyDict::new(py);
34        	kwargs.set_item("id", id);
35        	kwargs.set_item("backend", backend);
36			kwargs.set_item("qcs", qcs_pylist);
37			kwargs.set_item("shots", shots);
38			let running_result = module.call_method("run_qcs", (connection_str,), Some(&kwargs));
39			match running_result {
40				Ok(result) => result.unbind(),
41				Err(e) => {
42					// error!("Error running run_qc: {e}");
43					panic!("{e}, Error running run_qcs");
44				},
45			}
46		})
47	}
48}
